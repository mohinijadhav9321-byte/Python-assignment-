        "class SecureVault:\n",
        "    def __init__(self, pin):\n",
        "        self.__pin = pin\n",
        "        self.__data = {}\n",
        "\n",
        "    def add_entry(self, pin, key, value):\n",
        "        if pin == self.__pin:\n",
        "            self.__data[key] = value\n",
        "            print(\"Entry added successfully.\")\n",
        "        else:\n",
        "            print(\"Access Denied\")\n",
        "\n",
        "    def get_entry(self, pin, key):\n",
        "        if pin == self.__pin:\n",
        "            return self.__data.get(key)\n",
        "        else:\n",
        "            print(\"Access Denied\")\n",
        "\n",
        "\n",
        "# Creating object\n",
        "vault = SecureVault(\"1234\")\n",
